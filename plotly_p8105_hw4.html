The code below imports the restaurant inspection data.

library(p8105.datasets)
data("rest_inspec")

set.seed(1)

# make dataset smaller and easier to work with 
rest_inspec =
  rest_inspec %>% 
  filter(action == "Violations were cited in the following area(s).") %>% 
  drop_na() %>% 
  select(-phone, -camis)
  
rest_inspec = mutate(rest_inspec, year = year(rest_inspec$inspection_date)) %>% 
  sample_n(5000)

# plot number of critical violations over time
rest_inspec %>% 
  group_by(boro, critical_flag, year) %>% 
  summarize(n = n()) %>% 
  filter(critical_flag == "Critical") %>% 
  plot_ly(x = ~year, y = ~n, type = "scatter", mode = "line",
          alpha = 0.5,
          color = ~boro) %>% 
  ungroup() %>% 
  layout(xaxis = x, yaxis = y)

# number of A B C D E grades

  group_by(boro, grade) %>% 
  summarize(n = n()) %>% 
  mutate(grade = as.factor(grade)) %>% 
  plot_ly(x = ~boro, y = ~n, color = ~grade,
